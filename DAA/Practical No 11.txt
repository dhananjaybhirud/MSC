Practical No: - 11 Write a program to find longest common subsequence. 

#include<iostream>  
#include<conio.h>  
#include<string.h>  
using namespace std; 
int max(int a,int b); 
int lcs(char*X,char*Y,int m,int n) 
{ 
if(m==0||n==0)  
 return 0; 
if(X[m-1]==Y[n-1]) 
return 1+lcs(X,Y,m-1,n-1);  
else 
return max(lcs(X,Y,m,n-1),lcs(X,Y,m-1,n)); 
} 
int max(int a,int b) 
{ 
return(a>b)?a:b; 
} 
void main() 
{ 
char X[]="AGGTAB";  
char Y[]="GXTXAYB"; 
//clrscr(); 
int m = strlen(X);  
int n = strlen(Y); 
cout<<"Length of LCS is :- "<<lcs(X,Y,m,n);  
getch(); 
}
