void Allpaths::All_paths() 
{ 
for(i=1;i<=n;i++)  
 for(j=1;j<=n;j++)  
  A[i][j]=COST[i][j]; 
for(k=1;k<=n;k++)  
 for(i=1;i<=n;i++)  
  for(j=1;j<=n;j++)

 A[i][j]=A[i][j]; 
else

} 
}
