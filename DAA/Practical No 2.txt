void Union(int i,int j) 
{ 
int temp = Tree[i].nParent = Tree[j].nParent;  
if(Tree[i].nParent> Tree[j].nParent) 
{ 
Tree[i].nParent = j;  
Tree[j].nParent = temp; 
} 
else 
{ 
Tree[j].nParent = i;  
Tree[i].nParent = temp; 
} 
} 
int Find(int i) 
{ 
while(Tree[i].nParent>=0) 
{ 
i=Tree[i].nParent; 
} 
return i; 
}
