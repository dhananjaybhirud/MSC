#include<iostream>  
#include<conio.h>  
using namespace std; 
class SMATRIX 
{ 
int i,j,A[20][20],B[20][20],C[20][20],P,Q,R,S,T,U,V; 
public: 
void getdata();  
void SMATRIX1(); 
}; 
void SMATRIX::getdata() 
{ 
cout<<"\n Enter the 1st 2*2 matrix: -\n";  
for(i=1;i<=2;i++) 
{ 
for(j=1;j<=2;j++) 
{ 
cin>>A[i][j]; 
} 
} 
cout<<"Enter the 2nd 2*2 matrix: - \n";  
for(i=1;i<=2;i++) 
{ 
for(j=1;j<=2;j++) 
{ 
cin>>B[i][j]; 
} 
} 
} 
void SMATRIX::SMATRIX1() 
{ 
P=(A[1][1]+A[2][2])*(B[1][1]+B[2][2]); 
Q=(A[2][2]+A[2][2])*(B[1][1]); 
R=A[1][1]*(B[1][2]-B[2][2]); 
S=A[2][2]*(B[2][1]-B[2][2]); 
T=(A[1][1]+A[1][2])*B[2][2]; U=(A[2][2]-A[1][1])*(B[1][1]+B[1][2]); 



V=(A[1][2]-A[2][2])*(B[2][1]+B[2][2]); 
C[1][1]=(P+S)-(T-V);  
C[1][2]=R+T; 
C[2][1]=Q+S;  
C[2][2]=(P+R)-(Q+U); 
cout<<"\n Matrix Multiplication is :- \n";  
for(i=1;i<=2;i++) 
{ 
for(j=1;j<=2;j++) 
{ 
cout<<"\t"<<C[i][j]; 
} 
cout<<endl; 
} 
} 
void main() 
{ 
//clrscr(); 
SMATRIX s; 
s.getdata(); 
s.SMATRIX1(); 
getch(); 
}
