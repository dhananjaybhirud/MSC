{

};

{

for(i=0;i<n;i++) 
{ 
cin>>node[i];

}

for(i=0;i<n;i++) 
{ 
for(j=0;j<n;j++) 
{ 
cin>>a[i][j]; 
} 
} 
}

{

} 
} 
}
