img1=cv2.imread("E:\\Msc\\image.jpg",0) 
cv2.imshow("Gray Image",img1) 
 
 

 
 
kernal=np.ones((5,5),np.uint8) 
erosion=cv2.erode(img1,kernal,iterations=1) 
dilation=cv2.dilate(img1,kernal,iterations=1) 
opening=cv2.morphologyEx(img1,cv2.MORPH_OPEN,kernal) 
closing=cv2.morphologyEx(img1,cv2.MORPH_CLOSE,kernal) 
cv2.imshow('Erosion',erosion) 
 
 

 
 
cv2.imshow('Dilation',dilation) 
 
 

 
cv2.imshow('opening',opening) 
 
 

 
 
cv2.imshow('closing',closing)
